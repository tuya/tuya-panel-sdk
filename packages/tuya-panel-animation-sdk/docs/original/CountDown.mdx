`CountDown` 是倒计时动画组件。

## 代码演示

### 基础使用

<Playground>
  {() => {
    const [state, set] = React.useState({ countdownShow: 0, countDownLeft: 0 });
    const setState = value => set({ ...state, ...value });
    return (
      <View style={{ height:'100%', alignItems: 'center', justifyContent: 'center' }}>
        <CountDown
          countDownShow={state.countdownShow}
          countdownLeft={state.countdownLeft}
          // duration={400}
          // style={{
          //   backgroundColor: '#000',
          //   width: 120,
          //   height: 120,
          //   paddingTop: 40,
          //   paddingBottom: 40,
          // }}
          // activeBgColor="#000"
          // color="blue"
          // size={{ width: 30, height: 30 }}
          startCountdown={() => {
            setState({ countdownShow: true, countdownLeft: 300 });
          }}
          endCountdown={() => {
            setState({ countdownShow: false, countdownLeft: 0 });
          }}
        />
      </View>
    )
  }}
</Playground>
